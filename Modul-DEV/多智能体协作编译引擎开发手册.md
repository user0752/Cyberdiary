## 一、项目概述

### 1.1 背景与问题

赛博笔记（CyberNote）当前编译流程为 **Memo -> 单次 LLM 调用 -> WikiPage**，痛点：

| 问题 | 描述 |
|------|------|
| **质量瓶颈** | 单次 LLM 调用无法处理复杂的知识整理任务 |
| **角色缺失** | 没有分工，一个 Agent 既要提取实体又要写文章 |
| **无迭代机制** | 缺少评审-修订-优化的闭环 |
| **关联薄弱** | 缺少专门角色发现 Wiki 间的隐性关联 |
| **不可扩展** | 无法利用多个 LLM 模型各自的优势 |

### 1.2 解决方案

将编译从**单点调用**升级为 **AI 团队协作**，引入 **9 种 Agent 角色**（运行时最多 11 个实例），按 **5 层层次协作模式**编排。

角色清单：Coordinator（协调者）、Researcher x3（研究员，多视角并行）、Integrator（整合者）、Writer（写作者）、Reviewer x2（评审员，多维度并行）、Arbiter（仲裁者）、Editor（编辑）、Linker（链接发现者）。

### 1.3 核心目标

| 维度 | 目标 | 量化指标 |
|------|------|----------|
| **质量** | Wiki 质量提升 40%+ | BERTScore、ROUGE-L、结构完整性、可读性、信息密度 5 维评分 |
| **覆盖** | 隐性关联发现率 > 60% | 语义链接 / 总潜在链接 |
| **速度** | 编译耗时 < 2min（10 条 Memo） | 端到端延迟（单次 LLM 10~30s，并行研究 2~5s） |
| **可靠性** | 系统可用性 > 99.5% | 熔断恢复时间 < 5min |
| **可解释** | 编译过程全透明 | 审计日志 + 实时追踪面板 |

### 1.4 技术选型

| 组件 | 选择 | 理由 |
|------|------|------|
| **框架** | LangGraph | 原生 StateGraph、条件边、Send 并行 API |
| **LLM 路由** | LiteLLM（已有） | 统一接口，支持 DeepSeek / 阿里百炼 / Ollama |
| **并行** | asyncio.gather + Send | 异步 I/O 密集型 |
| **缓存** | SQLite + 内存 LRU | 轻量，无额外依赖 |
| **评估** | BERTScore + ROUGE-L + 结构分析 | 语义 + 文本重叠 |
| **前端** | Vue 3 + force-graph | 与项目现有选型一致 |

## 三、层次协作模式详解

### 3.1 L0 — 协调层（Coordinator）

```python
# backend/app/agents/coordinator_agent.py

from typing import List, Dict
import asyncio, os
from sklearn.cluster import KMeans
from litellm import aembedding

class CoordinatorAgent:
    """协调者 Agent — 多智能体系统的入口与出口"""

    async def run(self, state):
        memo_ids = state["memo_ids"]
        embedding_model = state["compilation_config"].get("embedding_model", "")

        if embedding_model and len(memo_ids) >= 3:
            state["memo_groups"] = await self.cluster_memos(memo_ids, embedding_model,
                n_clusters=min(3, len(memo_ids) // 2 + 1))
        else:
            state["memo_groups"] = [memo_ids]

        state["compilation_log"].append({
            "agent": "coordinator", "action": "clustered",
            "groups": [{"id": i, "size": len(g)} for i, g in enumerate(state["memo_groups"])]
        })
        state["current_layer"] = "research"
        return state

    async def cluster_memos(self, memo_ids, embedding_model, n_clusters):
        if n_clusters <= 1:
            return [memo_ids]
        embeddings = await self._batch_embed(memo_ids, embedding_model)
        kmeans = KMeans(n_clusters=n_clusters, random_state=42)
        labels = kmeans.fit_predict(embeddings)
        groups = [[] for _ in range(n_clusters)]
        for i, label in enumerate(labels):
            groups[label].append(memo_ids[i])
        return [g for g in groups if g]

    async def _batch_embed(self, memo_ids, embedding_model):
        """并行批量生成 Embedding。自动从编译配置的 model_config 提取 API key。"""
        from app.core.database import async_session
        from app.core.security import decrypt_api_key
        from app.models.memo import Memo
        from sqlalchemy import select

        async with async_session() as db:
            result = await db.execute(select(Memo.content).where(Memo.id.in_(memo_ids)))
            contents = [row[0] for row in result.all()]

        async def embed_one(content):
            resp = await aembedding(model=embedding_model, input=content[:8000],
                api_key=os.getenv("EMBEDDING_API_KEY") or os.getenv("DEEPSEEK_API_KEY") or "")
            return resp.data[0]["embedding"]

        return await asyncio.gather(*[embed_one(c) for c in contents])
```

> **设计说明**：Coordinator 聚类结果写入 `memo_groups` 用于日志记录和 `compilation_log`；实际的并行维度由 L1 的 Researcher 按固定 3 视角执行。当未配置 embedding_model 时跳过聚类。

```python
async def integrator_agent(state):
    import json, logging
    from app.services import llm_service
    logger = logging.getLogger(__name__)

    response = await llm_service.chat_completion(
        state["_model_config"],
        messages=[{"role":"system","content":INTEGRATOR_PROMPT.format(
```

```python
    from app.services import llm_service
```

### 3.5 L4 — 优化层（Editor + Linker）

```python
EDITOR_PROMPT = """你是一位内容编辑。根据评审反馈修订 Wiki。

## 原始 Wiki\n{wiki_content}\n\n## 评审意见\n{feedback}\n\n## 改进建议\n{suggestions}

## 要求
1. 针对每个问题逐一修改 2. 保留原文优点 3. 不得引入原文没有的新信息

## 输出
直接输出修订后的 Markdown 文章。
"""

async def editor_agent(state):
    from app.services import llm_service

    state["revision_count"] += 1
    if state["revision_count"] > state["compilation_config"].get("max_revisions", 3):
        state["review_passed"] = True
        # 路由由 _decide_next 统一决策
        return state

    response = await llm_service.chat_completion(
        state["_model_config"],
        messages=[{"role":"system","content":EDITOR_PROMPT.format(
            wiki_content=state.get("wiki_revised",state["wiki_draft"]),
            feedback=state["arbitration_result"]["summary"],
            suggestions="\n".join(state["arbitration_result"].get("priority_suggestions",[])))}],
        timeout=60
    )
    state["wiki_revised"] = response.choices[0].message.content
    state["current_agent"] = "reviewer_accuracy"
    return state


LINKER_PROMPT = """你是一位知识链接发现者。分析当前 Wiki 找出应与哪些已有 Wiki 建立关联。

## 当前 Wiki\n{current_wiki}\n\n## 已有 Wiki 列表\n{existing_wikis}

## 链接类型
- related_to / prerequisite / extended_by / contradicted_by / referenced_by

## 输出（JSON）
{{"suggested_links":[{{"target_slug":"existing-slug","relation_type":"related_to","confidence":0.85,"reason":"原因"}}]}}
"""


async def _get_existing_wikis_summary(state) -> list[dict]:
    """获取已有 Wiki 的摘要列表（供 Linker 发现关联用）"""
    from app.core.database import async_session
    from sqlalchemy import select

    async with async_session() as db:
        try:
            from app.models.wiki import WikiPage
            result = await db.execute(
                select(WikiPage.slug, WikiPage.title, WikiPage.summary).limit(50)
            )
            return [{"slug": r[0], "title": r[1], "summary": r[2] or ""} for r in result.all()]
        except Exception:
            return []


async def linker_agent(state):
    import json, logging
    from app.services import llm_service
    logger = logging.getLogger(__name__)

    existing = await _get_existing_wikis_summary(state)
    if not existing:
        state["suggested_links"] = []
        state["current_agent"] = "coordinator"
        return state

    response = await llm_service.chat_completion(
        state["_model_config"],
        messages=[{"role":"system","content":LINKER_PROMPT.format(
            current_wiki=state.get("wiki_revised",state["wiki_draft"]),
            existing_wikis=json.dumps(existing,ensure_ascii=False))}],
        response_format={"type":"json_object"}
    )
    try:
        raw = response.choices[0].message.content.strip()
        if raw.startswith("```"): raw = raw.split("\n",1)[1].rsplit("\n```",1)[0]
        result = json.loads(raw)
    except (json.JSONDecodeError,KeyError,AttributeError) as e:
        logger.warning("Linker JSON parse failed: %s", e)
        result = {"suggested_links":[]}

    state["suggested_links"] = result.get("suggested_links",[])
    state["current_agent"] = "coordinator"
    return state
```

## 四、LangGraph 状态图编排

### 4.1 状态定义

```python
# backend/app/models/agent_state.py

from typing import TypedDict, List, Dict, Optional, Literal, Any, Annotated
import operator
from enum import Enum

class AgentRole(str, Enum):
    COORDINATOR="coordinator"; RESEARCHER="researcher"; INTEGRATOR="integrator"
    WRITER="writer"; REVIEWER_ACCURACY="reviewer_accuracy"
    REVIEWER_READABILITY="reviewer_readability"; ARBITER="arbiter"
    EDITOR="editor"; LINKER="linker"

class CompilationState(TypedDict, total=False):
```

```python
        wf.add_node("integrator", integrator_agent)
        # L2

        # L3

        wf.add_node("reviewer_readability", reviewer_readability_agent)
        wf.add_node("arbiter", arbiter_agent)
        # L4
        wf.add_node("editor", editor_agent)
        wf.add_node("linker", linker_agent)

        wf.set_entry_point("coordinator")
        wf.add_edge("coordinator","researcher")
        wf.add_edge("researcher","integrator")
        wf.add_edge("integrator","writer")
        wf.add_edge("writer","reviewer_accuracy")
        wf.add_edge("writer","reviewer_readability")
        wf.add_edge("reviewer_accuracy","arbiter")
        wf.add_edge("reviewer_readability","arbiter")

        wf.add_conditional_edges("arbiter", self._decide_next,
            {"revise":"editor","link":"linker","finish":END})

        wf.add_edge("editor","reviewer_accuracy")
        wf.add_edge("editor","reviewer_readability")
        wf.add_edge("linker",END)

        return wf.compile(checkpointer=self.memory)

    def _decide_next(self, state):
        """统一路由决策——所有 Agent 不再自行设置 current_agent，由本函数集中判断。"""
        if state.get("next_action") == "finish": return "finish"
        if state.get("review_passed", False): return "link"
        if state.get("revision_count",0) > state.get("compilation_config",{}).get("max_revisions",3):
            return "link"
        return "revise"
```

### 5.2 Trace 数据模型

```python
# backend/app/core/compilation_tracer.py

import json, time, asyncio
from dataclasses import dataclass
from enum import Enum
from typing import Optional, Dict, List, Callable

class TraceEventType(str, Enum):
    PHASE_START="phase_start"; PHASE_END="phase_end"
    LLM_REQUEST="llm_request"; LLM_RESPONSE="llm_response"
    DECISION="decision"; WARNING="warning"; ERROR="error"; RESULT="result"

@dataclass
class TraceEntry:
    id: str; timestamp: float; agent: str; layer: str
    event_type: TraceEventType; title: str
    detail: Optional[str]=None; data: Optional[Dict]=None
    duration_ms: Optional[float]=None

class CompilationTracer:
    def __init__(self, job_id: str):
        self.job_id = job_id; self.entries = []
        self._seq = 0; self._phase_starts = {}
        self._sse_cb: Optional[Callable] = None

    def set_sse_callback(self, cb): self._sse_cb = cb

    def _emit(self, entry: TraceEntry):
        self.entries.append(entry)
        try:
            from app.services.compile_service import _compile_progress, _progress_lock
            with _progress_lock:
                p = _compile_progress.get(self.job_id,{})
                log = p.get("compilation_log",[])
                log.append({"agent":entry.agent,"layer":entry.layer,"event":entry.event_type.value,"title":entry.title})
                p["compilation_log"] = log; p["current_agent"] = entry.agent; p["current_layer"] = entry.layer
        except: pass
        if self._sse_cb:
            try: self._sse_cb(json.dumps({"event":"trace","data":{"id":entry.id,"timestamp":entry.timestamp,"agent":entry.agent,"layer":entry.layer,"event_type":entry.event_type.value,"title":entry.title,"detail":entry.detail,"data":entry.data,"duration_ms":entry.duration_ms}},ensure_ascii=False))
            except: pass

    def _next_id(self): self._seq+=1; return f"trace-{self._seq:04d}"

    def phase_start(self, agent, layer, title, detail=None):
        self._phase_starts[agent] = time.time()
        self._emit(TraceEntry(id=self._next_id(),timestamp=time.time(),agent=agent,layer=layer,event_type=TraceEventType.PHASE_START,title=title,detail=detail))

    def phase_end(self, agent, layer, title, data=None):
        start = self._phase_starts.pop(agent, time.time())
        self._emit(TraceEntry(id=self._next_id(),timestamp=time.time(),agent=agent,layer=layer,event_type=TraceEventType.PHASE_END,title=title,data=data,duration_ms=round((time.time()-start)*1000,1)))

    def llm_request(self, agent, layer, title, prompt_preview, model, full_prompt_len=0):
        self._emit(TraceEntry(id=self._next_id(),timestamp=time.time(),agent=agent,layer=layer,event_type=TraceEventType.LLM_REQUEST,title=title,detail=prompt_preview[:600],data={"model":model,"prompt_len":full_prompt_len or len(prompt_preview)}))

    def llm_response(self, agent, layer, title, preview, tokens=None):
        self._emit(TraceEntry(id=self._next_id(),timestamp=time.time(),agent=agent,layer=layer,event_type=TraceEventType.LLM_RESPONSE,title=title,detail=preview[:600],data={"tokens":tokens}))

    def decision(self, agent, layer, title, data):
        self._emit(TraceEntry(id=self._next_id(),timestamp=time.time(),agent=agent,layer=layer,event_type=TraceEventType.DECISION,title=title,data=data))

    def warning(self, agent, layer, title, detail=None):
        self._emit(TraceEntry(id=self._next_id(),timestamp=time.time(),agent=agent,layer=layer,event_type=TraceEventType.WARNING,title=title,detail=detail))

    def get_full_trace(self): return [{"id":e.id,"timestamp":e.timestamp,"agent":e.agent,"layer":e.layer,"event_type":e.event_type.value,"title":e.title,"detail":e.detail,"data":e.data,"duration_ms":e.duration_ms} for e in self.entries]
```

```python
import asyncio, json, logging
from app.services import llm_service

logger = logging.getLogger(__name__)

FOCUS_CONFIG = [
    {"name":"technical","label":"技术视角","prompt_extra":"关注技术细节、架构设计、代码实现"},

class ResearcherPool:
    def __init__(self, tracer=None, max_concurrency=3):
        self.tracer = tracer; self.semaphore = asyncio.Semaphore(max_concurrency)

    async def run_all(self, state):
        if self.tracer:
            self.tracer.phase_start("ResearcherPool","L1","启动多视角并行研究",f"{len(FOCUS_CONFIG)} 视角: 技术|实践|理论")

        results = await asyncio.gather(*tasks, return_exceptions=True)

            self.tracer.warning("ResearcherPool","L1",f"{failed}/{len(results)} Researcher 失败")
        if self.tracer:
            self.tracer.phase_end("ResearcherPool","L1","多视角研究完成",{"success":len(successful),"failed":failed,"entities":sum(len(r.get("entities",[])) for r in successful),"relations":sum(len(r.get("relations",[])) for r in successful)})
        return successful if successful else [{"entities":[],"relations":[],"key_topics":[],"_focus":"fallback"}]

    async def run_single(self, state, focus):
        async with self.semaphore:
            prompt = RESEARCHER_PROMPT.format(focus=focus["label"],focus_desc=focus["prompt_extra"],memos_content="\n\n---\n\n".join(state["memos_content"]))
            if self.tracer:
                self.tracer.llm_request(f"Researcher({focus['name']})","L1",f"LLM 请求 -- {focus['label']}",prompt[:600],state["compilation_config"]["model"],full_prompt_len=len(prompt))
            response = await llm_service.chat_completion(state["_model_config"],messages=[{"role":"system","content":prompt}],response_format={"type":"json_object"},timeout=30.0)
            try:
                raw = response.choices[0].message.content.strip()
                if raw.startswith("```"): raw = raw.split("\n",1)[1].rsplit("\n```",1)[0]
                result = json.loads(raw)
            except (json.JSONDecodeError,KeyError,AttributeError):
                logger.error("Researcher(%s) JSON parse failed",focus["name"])
                result = {"entities":[],"relations":[],"key_topics":[]}
            if self.tracer:
                preview = json.dumps(result,ensure_ascii=False)[:600]
                tokens = response.usage.total_tokens if response.usage else None
                self.tracer.llm_response(f"Researcher({focus['name']})","L1",f"LLM 响应 -- {focus['label']}",preview,tokens)
            result["_focus"] = focus["name"]
            return result
```

```python
@router.get("/jobs/{job_id}/stream")
async def stream_compile_progress(job_id: str, request: Request):
    from app.services.compile_service import _compile_progress
    from app.services.multi_agent_graph import _active_tracers

    # _progress_lock 由 compile_service 模块级声明（threading.Lock），
    # 所有 SSE 消费者 / Agent tracer 并发读写 _compile_progress 时持有此锁

    async def event_generator():
        trace_queue = asyncio.Queue()
        tracer = _active_tracers.get(job_id)
        if tracer:
            tracer.set_sse_callback(lambda msg: asyncio.ensure_future(trace_queue.put(msg)))
        last_progress = None
        while True:
            # 客户端断开时立即退出
            if await request.is_disconnected():
                break
            try:
                msg = trace_queue.get_nowait()
                yield f"data: {msg}\n\n"
                continue
            except asyncio.QueueEmpty: pass
            progress = _compile_progress.get(job_id)
            if not progress:
                yield f"data: {json.dumps({'event':'error','data':{'message':'Job not found'}})}\n\n"
                break
            if progress != last_progress:
                last_progress = dict(progress)
                s = progress.get("status")
                if s == "completed":
                    yield f"data: {json.dumps({'event':'complete','data':progress},ensure_ascii=False)}\n\n"; yield "data: [DONE]\n\n"; break
                elif s == "failed":
                    yield f"data: {json.dumps({'event':'error','data':progress},ensure_ascii=False)}\n\n"; break
                elif s == "needs_review":
                    yield f"data: {json.dumps({'event':'needs_review','data':progress},ensure_ascii=False)}\n\n"; break
                else:
                    yield f"data: {json.dumps({'event':'progress','data':progress},ensure_ascii=False)}\n\n"
            await asyncio.sleep(0.3)
    return StreamingResponse(event_generator(), media_type="text/event-stream")
```

### 7.2 熔断器

```python
# backend/app/core/circuit_breaker.py

import time
from enum import Enum

class CircuitState(Enum): CLOSED="closed"; OPEN="open"; HALF_OPEN="half_open"

class CircuitBreakerOpenError(Exception): pass

class CircuitBreaker:
    def __init__(self, name, failure_threshold=3, recovery_timeout=60.0):
        self.name = name; self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.state = CircuitState.CLOSED; self.failure_count = 0
        self.last_failure_time = 0.0

    async def call(self, func, *args, **kwargs):
        if self.state == CircuitState.OPEN:
            if time.time()-self.last_failure_time > self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
            else:
                raise CircuitBreakerOpenError(f"CB [{self.name}] OPEN")
        try:
            result = await func(*args, **kwargs)
            self.failure_count = 0
            if self.state == CircuitState.HALF_OPEN:
                self.state = CircuitState.CLOSED
            return result
        except Exception:
            self.failure_count += 1; self.last_failure_time = time.time()
            if self.failure_count >= self.failure_threshold:
                self.state = CircuitState.OPEN
            raise
```

```python
# backend/app/core/agent_error_handler.py

import logging; from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type
from litellm.exceptions import Timeout, RateLimitError, APIConnectionError

logger = logging.getLogger(__name__)

class AgentErrorHandler:
    def __init__(self):
        self.circuit_breakers = {k:CircuitBreaker(k) for k in ["researcher","writer","reviewer","editor","linker"]}

    async def execute(self, agent_name, func, state, fallback_value=None):
        cb = self.circuit_breakers.get(agent_name)
        if cb:
            try: return await cb.call(func, state)
            except CircuitBreakerOpenError:
                logger.warning("%s 熔断，使用降级", agent_name)
                return self._fallback(agent_name, state, fallback_value)
        return await self._retry(func, state)

    @retry(stop=stop_after_attempt(3),wait=wait_exponential(multiplier=1,min=1,max=10),
           retry=retry_if_exception_type((Timeout,RateLimitError,APIConnectionError)))
    async def _retry(self, func, state): return await func(state)

    def _fallback(self, agent, state, fb):
        if fb is not None: return fb
        if agent == "researcher": return [{"entities":[],"relations":[],"key_topics":[]}]
        if agent == "reviewer": return {"score":8.0,"feedback":"auto-pass"}
        if agent == "linker": return {"suggested_links":[]}
        return state
```

### 7.4 降级模型切换

```python
async def try_fallback_model(func, state, primary_model, fallback_model, db=None):
    """主模型失败时切换降级模型。注意：修改 compilation_config 副本 + 重建 _model_config，
    确保降级后的 LLM 调用使用正确的 API key / endpoint。"""
    import copy
    from app.services import llm_service
    config = copy.deepcopy(state.get("compilation_config", {}))

    # 尝试主模型（使用现有 _model_config）
    config["model"] = primary_model
    local_state = {**state, "compilation_config": config}
    try: return await func(local_state)
    except Exception as e: logger.warning("主模型 %s 失败: %s", primary_model, e)

    # 降级模型 —— 必须重建 _model_config（含新模型的 API key / endpoint）
    config["model"] = fallback_model
    new_mc = None
    if db:
        try: new_mc = await llm_service.get_model_config_from_db(db, fallback_model)
        except Exception: pass
    local_state = {**state, "compilation_config": config,
                   "_model_config": new_mc or state["_model_config"]}
    try: return await func(local_state)
    except Exception as e:
        logger.error("备用模型 %s 也失败: %s", fallback_model, e)
        return state
```

```python
            await asyncio.wait_for(wait_event.wait(), timeout=self.HUMAN_REVIEW_TIMEOUT)
        except asyncio.TimeoutError:
            logger.warning("人工审核超时，自动批准")
            review_task.status = "approved"
            review_task.final_content = review_task.wiki_content
            review_task.decided_at = datetime.now()
            await db.commit()
            self._wait_events.pop(state["job_id"], None)
            return "approved"
        await db.refresh(review_task)
        return review_task.status

    async def submit_decision(self, db, task_id, decision, edited_content=None):
        task = await db.get(HumanReviewTask, task_id)
        if not task: raise ValueError(f"Review task {task_id} not found")
        task.status = decision; task.decided_at = datetime.now()
        if decision == "approve": task.final_content = edited_content or task.wiki_content
        elif decision == "revise": task.user_edited_content = edited_content or task.wiki_content
        await db.commit()
        job_id = task.job_id
        if job_id in self._wait_events: self._wait_events[job_id].set()
        return {"review_passed":decision=="approve","next_action":"link" if decision=="approve" else "revise" if decision=="revise" else "finish"}

    async def _notify_frontend(self, task_id, state):
        from app.services.compile_service import _compile_progress
        job_id = state.get("job_id")
        if job_id and job_id in _compile_progress:
            _compile_progress[job_id].update({"status":"needs_review","review_task_id":task_id,"message":"等待人工审核..."})
```

```python
# backend/app/core/llm_cache.py

import hashlib, json, os, time, aiosqlite
from collections import OrderedDict

class LLMCache:
    def __init__(self, db_path="./data/llm_cache.db", memory_size=100, ttl=86400):
        self.memory = OrderedDict(); self.memory_size = memory_size; self.ttl = ttl
        self.db_path = os.path.expanduser(db_path); self.stats = {"mem":0,"disk":0,"miss":0}

    async def _init_db(self):
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)
        self._conn = await aiosqlite.connect(self.db_path)
        await self._conn.execute("CREATE TABLE IF NOT EXISTS llm_cache(cache_key TEXT PRIMARY KEY,prompt_hash TEXT,model TEXT,response TEXT,created_at REAL)")
        await self._conn.execute("CREATE INDEX IF NOT EXISTS idx_cache_ts ON llm_cache(created_at)")
        await self._conn.commit()

    async def _ensure_db(self):
        need_init = False
        if not hasattr(self, '_conn') or self._conn is None:
            need_init = True
        else:
            try:
                await self._conn.execute("SELECT 1")
            except Exception:
                need_init = True
        if need_init:
            await self._init_db()

    async def get(self, prompt, model, **kw):
        await self._ensure_db()
        key = hashlib.md5(f"{prompt}|{model}|{json.dumps(kw,sort_keys=True)}".encode()).hexdigest()
        if key in self.memory:
            v,ts = self.memory[key]
            if time.time()-ts < self.ttl: self.memory.move_to_end(key); self.stats["mem"]+=1; return v
            del self.memory[key]
        cur = await self._conn.execute("SELECT response,created_at FROM llm_cache WHERE cache_key=?",(key,))
        row = await cur.fetchone()
        if row and time.time()-row[1] < self.ttl:
            self.memory[key] = (row[0],row[1]); self.stats["disk"]+=1; return row[0]
        self.stats["miss"]+=1; return None

    async def set(self, prompt, model, response, **kw):
        await self._ensure_db()
        key = hashlib.md5(f"{prompt}|{model}|{json.dumps(kw,sort_keys=True)}".encode()).hexdigest()
        ts = time.time(); ph = hashlib.sha256(prompt.encode()).hexdigest()
        self.memory[key] = (response,ts)
        # LRU 淘汰：超过 memory_size 时移除最旧条目
        while len(self.memory) > self.memory_size:
            self.memory.popitem(last=False)
        await self._conn.execute("INSERT OR REPLACE INTO llm_cache VALUES(?,?,?,?,?)",(key,ph,model,response,ts))
        await self._conn.commit()
```

# backend/app/services/evaluation_service.py

import re, logging
logger = logging.getLogger(__name__)

class WikiEvaluationService:
    async def evaluate(self, wiki, reference=None, memos=None):
        scores = {"structure":self._eval_struct(wiki),"readability":self._eval_read(wiki)}
        if memos: scores["info_density"] = self._eval_density(wiki, memos)
        if reference:
            scores["bertscore"] = await self._bertscore(wiki, reference)
            scores["rouge_l"] = self._rouge_l(wiki, reference)
        scores["overall"] = sum(scores.values())/len(scores) if scores else 0
        return scores

    def _eval_struct(self, c):
        s = sum([0.25 if re.search(r'^# ',c,re.MULTILINE) else 0, 0.20 if re.search(r'^## ',c,re.MULTILINE) else 0, 0.15 if re.search(r'^- ',c,re.MULTILINE) else 0, 0.15 if c.count('```')>=2 else 0, 0.10 if '> ' in c else 0, 0.15 if re.search(r'\[.*\]\(.*\)',c) else 0])
        return min(s,1.0)

    def _eval_read(self, c):
        try: import jieba
        except ImportError: return 0.5
        sents = re.split(r'[。！？\n]',c); words = list(jieba.cut(c))
        if not sents or not words: return 0
        avg = len(words)/max(len([x for x in sents if x.strip()]),1)
        return max(0,1.0-abs(avg-20)/20)

    def _eval_density(self, wiki, memos):
        try: import jieba
        except ImportError: return 0.5
        ww = set(jieba.cut(wiki)); mw = set(jieba.cut(" ".join(memos)))
        return min(len(ww&mw)/max(len(mw)*0.1,1),1.0) if mw else 0

    async def _bertscore(self, c, r):
        try: from bert_score import score as bs; _,_,F1 = bs([c],[r],lang="zh",verbose=False); return float(F1.mean().item())
        except ImportError: logger.warning("bert_score not installed"); return 0.0

    def _rouge_l(self, c, r):
        try: from rouge_score import rouge_scorer; return rouge_scorer.RougeScorer(["rougeL"],use_stemmer=False).score(r,c)["rougeL"].fmeasure
        except ImportError: logger.warning("rouge_score not installed"); return 0.0

```vue
    .replace(/^## (.+)$/gm, '<h2>$1</h2>')
    .replace(/^# (.+)$/gm, '<h1>$1</h1>')
    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
    .replace(/\*(.+?)\*/g, '<em>$1</em>')
    .replace(/`(.+?)`/g, '<code>$1</code>')
    .replace(/\[\[(.+?)\]\]/g, '<a href="/wiki/$1" class="wiki-link">$1</a>')
    .replace(/\n\n/g, '</p><p>')
    .replace(/^- (.+)$/gm, '<li>$1</li>')
})

function formatDuration(ms?: number): string {
  if (!ms) return 'N/A'
  return ms < 1000 ? `${Math.round(ms)}ms` : `${(ms/1000).toFixed(1)}s`
}
</script>
```

## 十六、向后兼容与迁移方案

- **双模式编译**：`compile_type="single"` 走 `_do_compile()`，`compile_type="multi_agent"` 走 `run_multi_agent_compile()`
- **数据兼容**：两种模式输出到同一 `wiki_pages` 表；多智能体额外写入 `semantic_links`
- **协调时机**：`wiki_links` 在 `save_wiki_page` 内部写入（解析 `[[...]]` + tag fallback）；`semantic_links` 在 Linker 完成后、同一 DB 事务中写入
- **逐步迁移**：Phase 1 并存 -> Phase 2 A/B 验证 -> Phase 3 默认 multi -> Phase 4 移除 single
