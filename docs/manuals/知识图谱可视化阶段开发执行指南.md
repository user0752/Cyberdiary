#### 1.2 后端图谱数据聚合服务

**文件**：`backend/app/services/graph_data_service.py`

核心类 `GraphDataService`，职责：

- `get_knowledge_graph(job_id)` — 从编译结果中聚合完整图谱
  - 从 `CompilationState` 的 `integrated_knowledge.entities` 提取节点
  - 从 `integrated_knowledge.relations` 提取边
  - 从 `semantic_links` 表补充语义链接边
  - 节点去重：相同 label + type 合并，weight 取加权平均，memoCount 累加
  - 边去重：相同 source + target + type 合并，confidence 取最大值
  - 节点分组：优先使用 Coordinator 聚类结果，否则按 entity.type 分组
  - 返回 `KnowledgeGraph` 结构（nodes + edges + meta）

- `get_node_detail(job_id, node_id)` — 单节点详情 + 关联 edges + 关联 Wiki 列表

- `search_nodes(job_id, keyword)` — 模糊匹配 label

- `filter_subgraph(job_id, node_types, edge_types)` — 按类型筛选子图

**数据来源映射**：

| 数据源 | 位置 | 提取方式 |
|--------|------|----------|
| entities | `compile_jobs.compilation_log` 或运行时 `CompilationState["integrated_knowledge"]["entities"]` | JSON 解析 |
| relations | 同上 `["relations"]` | JSON 解析 |
| semantic_links | `semantic_links` 表 | SQL 查询 `WHERE source_slug IN (相关slug)` |
| memo 关联 | `memos` 表 | 通过 source_memo_ids 反查 |

**关键实现细节**：


```typescript
export async function searchGraphNodes(jobId: string, keyword: string): Promise<GraphNode[]> {
  const res = await client.get(`/compile/jobs/${jobId}/knowledge-graph/search`, { params: { q: keyword } })
  return res.data
}

// 按类型筛选
export async function filterKnowledgeGraph(
  jobId: string,
  types?: string[],
  edgeTypes?: string[]
): Promise<KnowledgeGraph> {
  const params: Record<string, string> = {}
  if (types?.length) params.types = types.join(',')
  if (edgeTypes?.length) params.edge_types = edgeTypes.join(',')
  const res = await client.get(`/compile/jobs/${jobId}/knowledge-graph/filter`, { params })
  return res.data
}

// SSE 图谱增量流
export async function* streamGraphUpdates(jobId: string): AsyncGenerator<{
  event: 'graph_update' | 'graph_complete';
  data: any;
}> {
  const response = await fetch(`/api/v1/compile/jobs/${jobId}/knowledge-graph/stream`)
  const reader = response.body!.getReader()
  const decoder = new TextDecoder()
  let buffer = ''

  while (true) {
    const { done, value } = await reader.read()
    if (done) break
    buffer += decoder.decode(value, { stream: true })
    const lines = buffer.split('\n')
    buffer = lines.pop() || ''
    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = line.slice(6)
        if (data === '[DONE]') return
        try {
          yield JSON.parse(data)
        } catch { /* skip malformed */ }
      }
    }
  }
}
```

#### 1.6 前端 Pinia Store

**文件**：`frontend/src/stores/knowledgeGraph.ts`

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { KnowledgeGraph, GraphNode, GraphEdge, NodeType, EdgeType } from '@/types/graph'
import { fetchKnowledgeGraph, fetchNodeDetail, searchGraphNodes } from '@/api/knowledgeGraph'

export const useKnowledgeGraphStore = defineStore('knowledgeGraph', () => {
  // 图谱数据
  const graph = ref<KnowledgeGraph | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)
  const currentJobId = ref<string | null>(null)

  // 交互状态
  const selectedNodeId = ref<string | null>(null)
  const hoveredNodeId = ref<string | null>(null)
  const searchKeyword = ref('')
  const searchResults = ref<GraphNode[]>([])

  // 筛选状态
  const activeNodeTypes = ref<Set<NodeType>>(new Set())
  const activeEdgeTypes = ref<Set<EdgeType>>(new Set())

  // 计算属性
  const filteredNodes = computed(() => {
    if (!graph.value) return []
    if (activeNodeTypes.value.size === 0) return graph.value.nodes
    return graph.value.nodes.filter(n => activeNodeTypes.value.has(n.type))
  })

  const filteredEdges = computed(() => {
    if (!graph.value) return []
    const nodeIds = new Set(filteredNodes.value.map(n => n.id))
    return graph.value.edges.filter(e => {
      const sourceId = typeof e.source === 'object' ? (e.source as any).id : e.source
      const targetId = typeof e.target === 'object' ? (e.target as any).id : e.target
      const typeOk = activeEdgeTypes.value.size === 0 || activeEdgeTypes.value.has(e.type)
      return typeOk && nodeIds.has(sourceId) && nodeIds.has(targetId)
    })
  })

  const stats = computed(() => ({
    totalNodes: filteredNodes.value.length,
    totalEdges: filteredEdges.value.length,
    nodeTypeCounts: graph.value?.meta.nodeTypes || {} as Record<NodeType, number>,
  }))

  // 动作
  async function loadGraph(jobId: string) {
    loading.value = true
    error.value = null
    currentJobId.value = jobId
    try {
      graph.value = await fetchKnowledgeGraph(jobId)
      // 初始化筛选：全选
      activeNodeTypes.value = new Set(graph.value.nodes.map(n => n.type))
      activeEdgeTypes.value = new Set(graph.value.edges.map(e => e.type))
    } catch (e: any) {
      error.value = e.message || 'Failed to load graph'
    } finally {
      loading.value = false
    }
  }

  function toggleNodeType(type: NodeType) {
    if (activeNodeTypes.value.has(type)) {
      activeNodeTypes.value.delete(type)
    } else {
      activeNodeTypes.value.add(type)
    }
  }

  function toggleEdgeType(type: EdgeType) {
    if (activeEdgeTypes.value.has(type)) {
      activeEdgeTypes.value.delete(type)
    } else {
      activeEdgeTypes.value.add(type)
    }
  }

  async function search(keyword: string) {
    searchKeyword.value = keyword
    if (!keyword.trim() || !currentJobId.value) {
      searchResults.value = []
      return
    }
    searchResults.value = await searchGraphNodes(currentJobId.value, keyword)
  }

  function selectNode(nodeId: string | null) {
    selectedNodeId.value = nodeId
  }

  function hoverNode(nodeId: string | null) {
    hoveredNodeId.value = nodeId
  }

  function resetFilters() {
    if (graph.value) {
      activeNodeTypes.value = new Set(graph.value.nodes.map(n => n.type))
      activeEdgeTypes.value = new Set(graph.value.edges.map(e => e.type))
    }
  }

  function $reset() {
    graph.value = null
    loading.value = false
    error.value = null
    currentJobId.value = null
    selectedNodeId.value = null
    hoveredNodeId.value = null
    searchKeyword.value = ''
    searchResults.value = []
    activeNodeTypes.value.clear()
    activeEdgeTypes.value.clear()
  }

  return {
    graph, loading, error, currentJobId,
    selectedNodeId, hoveredNodeId, searchKeyword, searchResults,
    activeNodeTypes, activeEdgeTypes,
    filteredNodes, filteredEdges, stats,
    loadGraph, toggleNodeType, toggleEdgeType, search,
    selectNode, hoverNode, resetFilters, $reset,
  }
})
```

```vue
  canvas: canvasRef, nodes: store.filteredNodes, edges: store.filteredEdges,
  transform, onNodeHover: store.hoverNode, onNodeSelect: store.selectNode,
  onTransformChange: requestRender,
})

// 渲染调度
let rafId = 0
function requestRender() {
  cancelAnimationFrame(rafId)
  rafId = requestAnimationFrame(() => {
    if (!canvasRef.value) return
    const ctx = canvasRef.value.getContext('2d')!
    const highlighted = store.hoveredNodeId ? getNeighbors(store.hoveredNodeId) : null
    render(ctx, {
      nodes: store.filteredNodes, edges: store.filteredEdges,
      transform: transform.value,
      hoveredNodeId: store.hoveredNodeId,
      highlightedNodeIds: highlighted?.nodeIds || new Set(),
      highlightedEdgeIds: highlighted?.edgeIds || new Set(),
    })
  })
}

// ResizeObserver
let resizeObs: ResizeObserver | null = null
onMounted(() => {
  if (containerRef.value) {
    resizeObs = new ResizeObserver(entries => {
      width.value = entries[0].contentRect.width
      height.value = entries[0].contentRect.height
      requestRender()
    })
    resizeObs.observe(containerRef.value)
  }
  setupListeners()
})

onUnmounted(() => {
  resizeObs?.disconnect()
  cleanup()
  dispose()
  cancelAnimationFrame(rafId)
})

// 数据变化时重新初始化布局
watch(() => store.filteredNodes, (nodes) => {
  init(nodes, store.filteredEdges)
  reheat()
}, { deep: false })
</script>
```

## 关键风险与应对

| 风险 | 影响 | 应对 |
|------|------|------|
| d3-force 与 Vue 响应式冲突 | 节点位置更新不触发重绘 | d3 就地修改 node 对象的 x/y，用 RAF 而非 Vue watch 驱动渲染 |
| Canvas 高 DPI 渲染模糊 | Retina 屏幕显示不清晰 | Canvas 尺寸按 devicePixelRatio 缩放，CSS 尺寸保持不变 |
| 大数据量性能瓶颈 | 500+ 节点卡顿 | 视口裁剪 + 离屏渲染缓存 + 大数据降级（隐藏标签、关闭 glow、30fps） |
| Three.js GPU 内存泄漏 | 3D↔2D 切换后内存增长 | 切换时手动 dispose geometry/material/texture |
| SSE 断连 | 编译过程中图谱停止更新 | 前端 EventSource onerror 自动重连 + 指数退避 |
| 编译数据中无 entities | 图谱为空 | 后端 fallback：从 wiki_pages 的 `[[双链]]` 和 tags 构建基础图谱 |
